Usage (cli)

```
echo 1658599410 | rjc timestamp
```

Schema

```
{
    naive: {
        year:         i32,
        month:        String,
        month_num:    u32,
        day:          u32,
        weekday:      String,
        weekday_num:  u32,
        hour:         u32,
        hour_24:      u32,
        minute:       u32,
        second:       u32,
        period:       String,
        day_of_year:  u32,
        week_of_year: u32,
        iso:          String,
    },
    utc: {
        year:         i32,
        month:        String,
        month_num:    u32,
        day:          u32,
        weekday:      String,
        weekday_num:  u32,
        hour:         u32,
        hour_24:      u32,
        minute:       u32,
        second:       u32,
        period:       String,
        utc_offset:   String,
        day_of_year:  u32,
        week_of_year: u32,
        iso:          String,
    }
}
```
